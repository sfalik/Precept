# Show available events and callable transitions from starting state
data
inspect

# data.LeftTurnQueued=true: Red advances to protected left-turn phase (FlashingGreen)
fire Advance

# Normal cycle continues from FlashingGreen (Green -> Yellow -> Red)
fire Advance
fire Advance
fire Advance

# Back at Red - no left-turn queue and no vehicle demand
inspect Advance
fire Advance

# Emergency without required args is rejected
fire Emergency

# Emergency with Emergency.AuthorizedBy and Emergency.Reason activates flashing red
fire Emergency '{"AuthorizedBy":"Dispatcher","Reason":"Accident"}'

# Advance has no transition while in emergency mode
fire Advance
state
data

# Clear the emergency to return to normal cycle
fire ClearEmergency
